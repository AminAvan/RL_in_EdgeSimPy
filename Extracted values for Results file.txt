import csv
import re

def extract_episode_content(file_path,x):
    # x = input("Enter the value of 'x' (1 to 3 digits): ")
    # if not x.isdigit() or not (1 <= int(x) <= 999):
    #     print("Invalid input! 'x' must be an integer with 1 to 3 digits.")
    #     return

    search_phrase = f"Episode {x} with"
    found = False
    extracted_content = []
    metrics = {
        "Hit-ratio": [],
        "runtime": [],
        "memory consumption": [],
        "power consumption": []
    }

    with open(file_path, 'r', encoding='utf-8') as csvfile:
        reader = csv.reader(csvfile)
        for row in reader:
            for cell in row:
                if found:  # If we're in the middle of capturing content
                    if '=' in cell:
                        end_index = cell.find('=')
                        extracted_content.append(cell[:end_index].strip())
                        found = False  # Stop capturing after '='
                        break
                    else:
                        extracted_content.append(cell.strip())
                elif search_phrase in cell:  # Start capturing content
                    found = True
                    start_index = cell.find(search_phrase) + len(search_phrase)
                    if '=' in cell[start_index:]:
                        end_index = cell.find('=', start_index)
                        extracted_content.append(cell[start_index:end_index].strip())
                        found = False  # Stop capturing immediately if '=' is in the same cell
                        break
                    else:
                        extracted_content.append(cell[start_index:].strip())

    # Process extracted content to find and store required metrics
    for line in extracted_content:
        # Match metrics using regex
        if "Hit-ratio:" in line:
            match = re.search(r"Hit-ratio: ([\d.]+)%", line)
            if match:
                metrics["Hit-ratio"].append(float(match.group(1)))
        elif "runtime:" in line:
            match = re.search(r"runtime: ([\d.]+) seconds", line)
            if match:
                metrics["runtime"].append(float(match.group(1)))
        elif "memory consumption:" in line:
            match = re.search(r"memory consumption: ([\d.]+) MB", line)
            if match:
                metrics["memory consumption"].append(float(match.group(1)))
        elif "power consumption:" in line:
            match = re.search(r"power consumption: ([\d.]+) Watt-seconds", line)
            if match:
                metrics["power consumption"].append(float(match.group(1)))

    # Calculate and display averages
    print(f"\nExtracted metrics for 'Episode {x}':")
    for key, values in metrics.items():
        if values:
            average = sum(values) / len(values)
            print(f"Average {key}: {average:.2f}")
        else:
            print(f"No data for {key}.")

  extract_episode_content(file_path,i)

import re

# Initialize a list to store the results
results = []

# Temporary variables to hold the values until a pair is formed
current_hit_ratio = None
current_power = None

# Read and process the file
with open(file_path, "r") as file:
    lines = file.readlines()
    for line in lines:
        # Match 'Average Hit-ratio'
        hit_match = re.search(r"Average Hit-ratio: ([\d.]+)", line)
        if hit_match:
            current_hit_ratio = float(hit_match.group(1))

        # Match 'Average power consumption'
        power_match = re.search(r"Average power consumption: ([\d.]+)", line)
        if power_match:
            current_power = float(power_match.group(1))

        # If both values are available, append the pair and reset
        if current_hit_ratio is not None and current_power is not None:
            results.append((current_power, current_hit_ratio))
            current_hit_ratio = None
            current_power = None
